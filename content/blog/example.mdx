export async function CodeBlock({
  children,
  lang,
  ...rest
}: HTMLAttributes<HTMLElement> & {
  children: string
  lang: string
}) {
  const rendered = await highlight(children, {
    lang,
    theme: "vesper",
    components: {
      pre: (props) => <Pre {...props} />,
    },
  })

  return <CodeBlockClient {...rest}>{rendered}</CodeBlockClient>
}
